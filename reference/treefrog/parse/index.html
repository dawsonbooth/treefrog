    class ParseError(IOError):

    def __init__(self, message, filename = None, pos = None):

        super().__init__(message)

        self.filename = filename

        self.pos = pos

    def __str__(self):

        return 'Parse error (%s %s): %s' % (

            self.filename or '?',

            '@0x%x' % self.pos if self.pos else '?',

            super().__str__())
